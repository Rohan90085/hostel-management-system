async function searchStudent() {
    const keyword = document.getElementById("searchText").value;

    if (!keyword) {
        alert("Please enter ID or Name");
        return;
    }

    const response = await fetch("/students/search", {
        method: "POST",
        headers: { "Content-Type": "application/json" },
        body: JSON.stringify({ keyword })
    });

    const data = await response.json();

    if (data.length === 0) {
        alert("No student found");
        return;
    }

    // Build message for alert
    let message = "Search Result(s):\n\n";

    data.forEach(s => {
        message += 
            `ID: ${s.id}\n` +
            `Name: ${s.name}\n` +
            `Phone: ${s.phone}\n` +
            `Gender: ${s.gender}\n\n`;
    });

    alert(message);
}
</script>
